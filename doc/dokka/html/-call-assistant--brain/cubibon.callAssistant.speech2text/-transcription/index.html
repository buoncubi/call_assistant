class Transcription(message: String = UNKNOWN_MESSAGE, confidence: Double = UNKNOWN_CONFIDENCE, startTime: Long = UNKNOWN_TIME, endTime: Long = UNKNOWN_TIME, sourceTag: String = ServiceInterface.UNKNOWN_SOURCE_TAG) : CallbackInput
